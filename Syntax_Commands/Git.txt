GIT Commands
------------

git init

git remote add origin <link>

git add .

git remote -v

git commit -m "comment"

git push origin <branch_name>

git pull origin <branch_name>

git status

git log

git revert -n <hash>
git revert HEAD

git reset <hash> --hard
git reset HEAD~1 --hard

git branch <branch_name>
git branch -d <branch_name>

git checkout <branch_name>

git checkout -b <new branch_name>

git merge

git diff

git clone

git stash pop

git rebash -i HEAD~3

git squash

git rm -r . --cached

git rev-parse HEAD  --> to get the latest commit SHA
